});

  try {
    // 1. Validar schema

    });

    // 3. Criar no DB
    const post = await prisma.post.create({
      data: {
        title: title.trim(),
        content: cleanContent,
        userId: req.userId,
      },
    });

    // 4. Award pontos (async, n√£o bloqueia)
    awardPoints({
      userId: req.userId,
      type: 'POST_CREATED',
      points: 10,
      meta: { postId: post.id },
    }).catch(err => console.error('Points award error:', err));

    return res.status(201).json(post);
  } catch (error: any) {
    return res.status(500).json({ error: error.message });
  }
});
